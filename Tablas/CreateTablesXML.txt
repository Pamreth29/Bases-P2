CREACION DE TABLAS:


-- Tabla Clientes (ClienteNuevo)
CREATE TABLE ClienteNuevo (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    NumIdentificacion INT NOT NULL,
    Nombre VARCHAR(64) NOT NULL
);

-- Tabla TipoTarifa
CREATE TABLE TipoTarifa (
    Id INT PRIMARY KEY,
    Nombre VARCHAR(64) NOT NULL
);

-- Tabla TipoUnidad
CREATE TABLE TipoUnidad (
    Id INT PRIMARY KEY,
    Nombre VARCHAR(64) NOT NULL
);

-- Tabla TipoElemento
CREATE TABLE TipoElemento (    
    Id INT PRIMARY KEY,
    Nombre VARCHAR(64) NOT NULL
);

-- Tabla TipoRelacionFamiliar
CREATE TABLE TipoRelacionFamiliar (
    Id INT PRIMARY KEY,
    Nombre VARCHAR(64) NOT NULL
);

-- Tabla ElementoDeTipoTarifa
CREATE TABLE ElementoDeTipoTarifa (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    idTipoTarifa INT FOREIGN KEY REFERENCES TipoTarifa(Id),
    IdTipoElemento INT FOREIGN KEY REFERENCES TipoElemento(Id),
    Valor INT NOT NULL,
    IdTipoUnidad INT FOREIGN KEY REFERENCES TipoUnidad(Id)
);

-- Tabla Contratos (NuevoContrato)
CREATE TABLE NuevoContrato (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    ID_Cliente INT FOREIGN KEY REFERENCES ClienteNuevo(Id),
    Numero NVARCHAR(100) NOT NULL,
    DocIdentidadCliente NVARCHAR(100) NOT NULL,
    TipoTarifa INT FOREIGN KEY REFERENCES TipoTarifa(Id),
    Fecha DATETIME NOT NULL
);

-- Tabla UsoDatos
CREATE TABLE UsoDatos (
    Id INT IDENTITY(1,1) PRIMARY KEY,
    Numero INT NOT NULL,
    QGigas FLOAT NOT NULL
);

-- Tabla TipoLlamada
CREATE TABLE TipoLlamada (
    Id INT PRIMARY KEY,
    isLocal BIT NOT NULL  -- 1 para true, 0 para false
);

-- Tabla Llamadas
CREATE TABLE Llamadas (
    Id INT PRIMARY KEY,
    ID_Contrato INT FOREIGN KEY REFERENCES NuevoContrato(Id),
    ID_TipoLlamada INT FOREIGN KEY REFERENCES TipoLlamada(Id),
    NumeroDe VARCHAR(64) NOT NULL,
    NumeroA VARCHAR(64) NOT NULL,
    Duracion INT NOT NULL,  -- Duraci√≥n en minutos
    FechaHoraInicio DATETIME NOT NULL,
    FechaHoraFin DATETIME NOT NULL,
    CostoTotal MONEY NOT NULL,
);
